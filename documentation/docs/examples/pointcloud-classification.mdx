// import { WellKnownAsprsPointClassCodes } from '@cognite/reveal'

const modelClasses = model.getClasses();

// specify classes you want to be displayed
const classesToShow = [
  WellKnownAsprsPointClassCodes.HighWegetation,
  WellKnownAsprsPointClassCodes.Ground,
];

// show only selected classes, hide all the others
modelClasses.forEach((classCode) => {
  model.setClassVisible(classCode, classesToShow.includes(classCode));
});
